<BilingualContent
  en={<h2>Code Example: Surface Classification</h2>}
  ur={<h2 style={{textAlign: 'right'}}>کوڈ مثال: سطح کی درجہ بندی</h2>}
/>

```python
"""
Train robot to recognize walking surfaces using scikit-learn
روبوٹ کو چلنے کی سطحوں کو پہچاننے کی ٹریننگ دیں
"""

import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score
import pickle

# Training data: accelerometer readings from different surfaces
# ٹریننگ ڈیٹا: مختلف سطحوں سے ایکسلرومیٹر ریڈنگز
# Format: [ax, ay, az] in g-force
data = np.array([
    [0.1, 0.2, 9.8],  # Tile sample 1 | ٹائل نمونہ 1
    [0.15, 0.18, 9.8],  # Tile sample 2 | ٹائل نمونہ 2
    [0.2, 1.5, 9.7],  # Carpet sample 1 | قالین نمونہ 1
    [0.25, 1.8, 9.6],  # Carpet sample 2 | قالین نمونہ 2
    # ... collect more samples from your robot ...
])

# Labels: 0=tile, 1=carpet, 2=wood
# لیبلز: 0=ٹائل، 1=قالین، 2=لکڑی
labels = np.array([0, 0, 1, 1, 2, 2])

# Split: 80% training, 20% testing
# تقسیم: 80% ٹریننگ، 20% ٹیسٹنگ
X_train, X_test, y_train, y_test = train_test_split(
    data, labels, test_size=0.2, random_state=42
)

# Train Random Forest | رینڈم فاریسٹ ٹرین کریں
model = RandomForestClassifier(n_estimators=10, random_state=42)
model.fit(X_train, y_train)

# Evaluate | جانچ
predictions = model.predict(X_test)
accuracy = accuracy_score(y_test, predictions)
print(f"Model Accuracy: {accuracy:.1%}")

# Save for deployment | تعیناتی کے لیے محفوظ کریں
with open('surface_classifier.pkl', 'wb') as f:
    pickle.dump(model, f)
```
